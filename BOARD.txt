CREATE TABLE `tbl_board` (
  `bno` int(11) NOT NULL,
  `title` varchar(500) NOT NULL,
  `content` text,
  `writer` varchar(45) NOT NULL,
  `regdate` timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `viewcnt` int(11) DEFAULT '0',
  PRIMARY KEY (`bno`)
);

 * 데이터베이스 관련
	- 계정, 디비 생성
	- 테이블 설계, 생성 (ERD)
	- 더미데이터 준비
	https://www.mockaroo.com/

 * 스프링MVC 관련
	- 패키지 구조 설계, 코드 규칙 작성
	- 개발 역할 분담
	- 테스트 동작, 기능 구현
	
 * 화면 관련
	-화면 방식 결정
	-경로 설정(절대 경로, 상대경로)	

 * 게시판 동작
 	동작					URI						방식		처리 후 페이지

 	글 수정하기				/board/modify?bon=x		GET			수정.jsp (view 연결)
 	글 수정하기(처리)		/board/modify			POST		리스트페이지/본문페이지(컨트롤러)
